함수의 인자값이 제네릭 타입인 경우 extends를 사용하여 타입을 제한할 수 있음

```js
function getPlayerName<T extends string>(playerName: T) {
  console.log(playerName);
}

getPlayerName(155); // error
getPlayerName("김도영"); // ok

interface Player {
  name: string;
  age: number;
}

function getPlayer<T extends Player>(playerName: T) {
  console.log(playerName);
}

getPlayer(155); // error
getPlayer("김도영"); // error
getPlayer({ name: "김도영", age: 20 }); // ok
getPlayer({ name: "김도영", age: 20, backNumber: 5 }); // ok
```
